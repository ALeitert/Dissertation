% QED symbol
\DeclareRobustCommand{\qed}
{%
    \def\qedChar{\square}%
    \ifnum 1 < \value{proofLevel}%
        \def\qedChar{\lozenge}%
    \fi%
    \ifmmode%
        \qedChar%
    \else%
        \leavevmode\unskip\penalty9999 \hbox{}\nobreak\hfill%
        \quad\hbox{$\qedChar$}%
    \fi%
}%
